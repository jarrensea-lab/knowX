    # 金叉：5 日线上穿 20 日线
    if row['MA5'] > row['MA20']:
        return True
    return False

def sell_signal(row):
    # 死叉：5 日线下穿 20 日线
    # 或收盘价跌破 20 日线的 92%
    if row['MA5'] < row['MA20'] or row['Close'] < row['MA20'] * 0.92:
        return True
    return False

## 章节 9：RSI 指标策略

### 9.1 RSI 计算
```python
def calculate_rsi(prices, period=14):
    delta = prices.diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=period).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=period).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    return rsi
```

### 9.2 RSI 交易信号
- **超买区**: RSI > 70，考虑卖出
- **超卖区**: RSI < 30，考虑买入
- **金叉/死叉**: RSI 穿越 50 中线

## 章节 14：Backtrader 回测框架

### 14.1 双均线策略
```python
import backtrader as bt

class DualMAStrategy(bt.Strategy):
    params = (
        ('fast_period', 10),
        ('slow_period', 30),
    )

    def __init__(self):
        self.fast_ma = bt.indicators.SMA(
            self.data.close, period=self.params.fast_period
        )
        self.slow_ma = bt.indicators.SMA(
            self.data.close, period=self.params.slow_period
        )

    def next(self):
        if not self.position:
            # 金叉买入
            if self.fast_ma > self.slow_ma:
                self.buy()
        else:
            # 死叉卖出
            if self.fast_ma < self.slow_ma:
                self.sell()
```

class FixedStopTake(bt.Strategy):
    params = (
        ('stop_loss_pct', 0.05),  # 止损 5%
        ('take_profit_pct', 0.10), # 止盈 10%
        ('ma_period', 20),
    )

    def next(self):
        if not self.position:
            if self.data.close > self.ma:
                self.buy()
                self.buy_price = self.data.close[0]
        else:
            stop_price = self.buy_price * 0.95
            take_profit_price = self.buy_price * 1.10
            if self.data.close < stop_price or self.data.close > take_profit_price:
